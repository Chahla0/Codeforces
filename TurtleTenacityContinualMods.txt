#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >>n;
    while (n--)
    {
        int t,s=0;
        cin>>t;
        int tt[t];
        for (size_t i = 0; i < t; i++)
        {s
            cin >>tt[i];
            if(tt[i]!=0)
            {
                s=s%tt[i];
            }
        }
        if(s==1)    cout <<"YES";
        else     cout <<"NO";
    }
    
}
